   130→/**
   131→ * Operator-preference card for the critical-alarm audible beep. The setting
   132→ * is stored per-browser in localStorage (no API, no sync) so each operator /
   133→ * kiosk can choose independently.
   134→ */
   135→function AlarmSoundCard() {
   136→  const [enabled, setEnabled] = useState<boolean>(true)
   137→
   138→  // Hydrate from localStorage after mount (SSR-safe).
   139→  useEffect(() => {
   140→    setEnabled(isAlarmSoundEnabled())
   141→  }, [])
   142→
   143→  const handleToggle = (next: boolean) => {
   144→    setEnabled(next)
   145→    setAlarmSoundEnabled(next)
   146→    if (next) {
   147→      toast.success('Critical alarm sound enabled', {
   148→        description: 'An audible 3-beep pattern will play when a critical alarm fires.',
   149→      })
   150→      // Preview the beep immediately so the operator can confirm their speakers work.
   151→      // Note: this runs inside the click handler — satisfies the autoplay gesture.
   152→      playCriticalBeep()
   153→    } else {
   154→      toast.info('Critical alarm sound muted', {
   155→        description: 'Visual banners and toasts remain. Recommended only for mobile browsing.',
   156→      })
   157→    }
   158→  }
   159→
   160→  const handleTest = () => {
   161→    playCriticalBeep()
   162→    toast.info('Test beep played', { description: 'If you did not hear it, check browser audio output.' })
   163→  }
   164→
   165→  return (
   166→    <div className="rounded-md border border-border/50 bg-card/40 p-3">
   167→      <div className="flex items-start justify-between gap-3">
   168→        <div className="flex items-start gap-2.5">
   169→          <Volume2 className={cn('mt-0.5 h-4 w-4', enabled ? 'text-amber-400' : 'text-muted-foreground')} />
   170→          <div className="min-w-0">
   171→            <p className="text-xs font-medium">Alarm Sound</p>
   172→            <p className="mt-0.5 text-[10px] leading-relaxed text-muted-foreground">
   173→              Play an audible beep when a critical alarm fires. Recommended for control rooms. Stored locally per browser.
   174→            </p>
   175→          </div>
   176→        </div>
   177→        <div className="flex shrink-0 items-center gap-2">
   178→          {enabled && (
   179→            <Badge variant="outline" className="h-5 px-1.5 text-[10px] text-amber-400 ring-amber-500/30">Enabled</Badge>
   180→          )}
   181→          <Switch checked={enabled} onCheckedChange={handleToggle} aria-label="Toggle critical alarm sound" />
   182→        </div>
   183→      </div>
   184→      <div className="mt-3">
   185→        <Button size="sm" variant="outline" className="h-7 gap-1.5 text-xs" onClick={handleTest}>
   186→          <Volume2 className="h-3 w-3" /> Test Sound
   187→        </Button>
   188→      </div>
   189→    </div>
   190→  )
   191→}
