    39→function ServiceCard({
    40→  icon: Icon, name, status = 'healthy', endpoint, version, onConfigure, accent = 'emerald',
    41→}: {
    42→  icon: any; name: string; status?: 'healthy' | 'degraded' | 'down'; endpoint?: string; version?: string; onConfigure?: () => void; accent?: 'emerald' | 'sky' | 'amber' | 'violet' | 'rose'

    45→  const statusLabel = status === 'healthy' ? 'Healthy' : status === 'degraded' ? 'Degraded' : 'Down'
    46→  const accentMap: Record<string, string> = {
    47→    emerald: 'text-emerald-400 bg-emerald-500/10 ring-emerald-500/20',
    48→    sky: 'text-sky-400 bg-sky-500/10 ring-sky-500/20',
    49→    amber: 'text-amber-400 bg-amber-500/10 ring-amber-500/20',
    50→    violet: 'text-violet-400 bg-violet-500/10 ring-violet-500/20',
    51→    rose: 'text-rose-400 bg-rose-500/10 ring-rose-500/20',
    52→  }
    53→  return (
    54→    <Card className="gap-0 p-4">
    55→      <div className="flex items-start justify-between">
    56→        <div className="flex items-center gap-2.5">
    57→          <div className={cn('rounded-md p-2 ring-1', accentMap[accent])}>
    58→            <Icon className="h-4 w-4" />

    63→              <span className={cn('inline-block h-1.5 w-1.5 rounded-full', dot)} /> {statusLabel}
